Code Snippet: Networked Player Movement


using UnityEngine;
using Unity.Netcode;

public class NetworkPlayerMovement : NetworkBehaviour
{
    public float moveSpeed = 5f;

    void Update()
    {
        if (!IsOwner) return; // Only control your own player

        // Example: Simple movement input
        float horizontal = Input.GetAxis("Horizontal");
        float vertical = Input.GetAxis("Vertical");

        Vector3 moveDirection = new Vector3(horizontal, 0, vertical).normalized;
        transform.position += moveDirection * moveSpeed * Time.deltaTime;

        // If this were a real game, you'd likely send input to the server
        // via a ServerRpc or use NetworkTransform.
    }
}

Example of a basic networked player movement script using Unity Netcode, ensuring only the owner controls their character.
